~[tlist_sql; WITH stuRx as ( SELECT u_fb_medications.studentsdcid, Count(u_fb_medications.name) cnt, '
    ' || listagg('
  • ' || u_fb_medications.name || CASE WHEN u_fb_medications.prescribed = 'Yes' THEN '*' ELSE '' END || ' - ' || u_fb_medications.dosage || ' ' || u_fb_medications.dosage_units || '| ' || u_fb_medications.frequency_taken || '
    • ', '') WITHIN GROUP (ORDER BY u_fb_medications.name desc) || '
' Rxs FROM u_fb_medications GROUP BY u_fb_medications.studentsdcid ) SELECT students.dcid, students.student_number, students.lastfirst, CASE WHEN UPPER(students.gender) = 'F' THEN 'Female' WHEN UPPER(students.gender) = 'M' THEN 'Male' ELSE students.gender END as gender, CASE WHEN TO_CHAR(students.grade_level) = -2 then 'PK3' WHEN TO_CHAR(students.grade_level) = -1 then 'PK4' WHEN TO_CHAR(students.grade_level) = 0 then 'K' WHEN TO_CHAR(students.grade_level) = 99 then 'Graduated' ELSE TO_CHAR(students.grade_level) END as PRETTY_GRADE, stuRx.cnt, stuRx.Rxs FROM students LEFT JOIN stuRx on students.dcid =stuRx.studentsdcid ~[if#ss.~(gpv.ss;sqlText)=cur] INNER JOIN ~[temp.table.current.selection:students] temp ON (students.dcid = temp.dcid) [/if#ss] WHERE 0=0 ~[if.is.a.school] AND students.schoolid = ~(curschoolid) [/if.is.a.school] ~[if#perscribed.~(gpv.perscribed;sqlText)=1] AND stuRx.cnt > 0 [/if#perscribed] ORDER BY students.lastfirst ;] [/tlist_sql]
