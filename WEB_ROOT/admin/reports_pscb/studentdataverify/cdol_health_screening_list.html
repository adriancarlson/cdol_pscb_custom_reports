~[tlist_sql; SELECT dcid, state_studentnumber, student_number, student_first_name, student_last_name, gender, dob, PRETTY_GRADE, teacher_name, ADDRESS, LISTAGG(Contact, ' | ') WITHIN GROUP(ORDER BY DCID,contactpriorityorder) as Adults_students_live_with, school FROM ( SELECT s.dcid, s.state_studentnumber, s.student_number, s.first_name as student_first_name , s.last_name as student_last_name, CASE WHEN UPPER(s.gender) = 'M' THEN 'Male' WHEN UPPER(s.gender) = 'F' THEN 'Female' ELSE s.gender END as GENDER, s.dob, CASE WHEN TO_CHAR(s.grade_level) = -2 then 'PK3' WHEN TO_CHAR(s.grade_level) = -1 then 'PK4' WHEN TO_CHAR(s.grade_level) = 0 then 'K' WHEN TO_CHAR(s.grade_level) = 99 then 'Graduated' ELSE TO_CHAR(s.grade_level) END as PRETTY_GRADE, t.title ||' ' || t.first_name || ' ' || t.last_name as teacher_name, s.street || ' ' || s.city || ' ' || s.state || ' ' || s.zip as ADDRESS, CASE WHEN p.firstname IS NULL AND p.lastname IS NULL THEN '-No Name-' ELSE p.firstname||' '||p.lastname ||' - '|| cs.code END as Contact, sca.contactpriorityorder contactpriorityorder, sch.name as school from cc inner join STUDENTS s on cc.STUDENTID = s.ID ~[if#ss.~(gpv.ss;sqlText)=cur] INNER JOIN ~[temp.table.current.selection:students] temp ON (s.dcid = temp.dcid) [/if#ss] inner join TEACHERS t on cc.TEACHERID = t.ID JOIN schools sch ON (s.schoolid = sch.school_number) LEFT JOIN studentcontactassoc sca ON (s.dcid = sca.studentdcid) LEFT JOIN studentcontactdetail scd ON (sca.studentcontactassocid = scd.studentcontactassocid) LEFT JOIN person p ON (sca.personid = p.ID) LEFT JOIN codeset cs ON (cs.codesetid = scd.relationshiptypecodesetid) LEFT JOIN guardianpersonassoc gp ON (p.id = gp.personid) LEFT JOIN guardianstudent gs ON (s.dcid = gs.studentsdcid AND gp.guardianid = gs.guardianid) WHERE s.enroll_status = 0 and cc.COURSE_NUMBER = '541001' and cc.TERMID BETWEEN ~(curyearid)00 AND ~(curyearid)99 AND scd.liveswithflg = 1 AND cs.code NOT IN ('Brother','Friend','Neighbor','Other','Sister','Resource Teacher') ~[if.is.a.school] AND s.schoolid = ~(curschoolid) [/if.is.a.school] ~[if#gr.~(gpv.gr;sqlText)#] AND s.grade_level = ~(gpv.gr) [/if#gr] ORDER BY s.lastfirst, sca.contactpriorityorder ) GROUP BY dcid, state_studentnumber, student_number, student_first_name, student_last_name, gender, dob, PRETTY_GRADE, teacher_name, ADDRESS, school ORDER BY GRADE_LEVEL, LASTFIRST;] [/tlist_sql]
