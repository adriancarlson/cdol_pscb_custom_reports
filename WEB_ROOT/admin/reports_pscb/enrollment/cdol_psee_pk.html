Preschool Enrollments

~[tlist_sql; WITH student_form AS ( SELECT students.id AS studentid, students.dcid, students.student_number, students.lastfirst, students.first_name, students.last_name, CASE WHEN UPPER(students.gender) = 'F' THEN 'Female' WHEN UPPER(students.gender) = 'M' THEN 'Male' ELSE students.gender END AS gender, students.dob, CASE WHEN TO_CHAR(students.grade_level) = -2 THEN 'PK3' WHEN TO_CHAR(students.grade_level) = -1 THEN 'PK4' WHEN TO_CHAR(students.grade_level) = 0 THEN 'K' WHEN TO_CHAR(students.grade_level) = 99 THEN 'Graduated' ELSE TO_CHAR(students.grade_level) END AS current_grade, CASE WHEN TO_CHAR(students.Sched_NextYearGrade) = -2 THEN 'PK3' WHEN TO_CHAR(students.Sched_NextYearGrade) = -1 THEN 'PK4' WHEN TO_CHAR(students.Sched_NextYearGrade) = 0 THEN 'K' WHEN TO_CHAR(students.Sched_NextYearGrade) = 99 THEN 'Graduated' ELSE TO_CHAR(students.Sched_NextYearGrade) END AS next_grade, students.schoolid AS current_school, students.next_school AS next_school, CASE WHEN students.Sched_NextYearGrade IN (-2,-1) THEN CASE WHEN preschool.approval = 0 THEN 'Pending' WHEN preschool.approval = 1 THEN 'Approved' WHEN preschool.approval = 2 THEN 'Rejected' WHEN preschool.response_type = 'A' THEN 'Approved' ELSE 'Not Submitted' END ELSE 'N/A' END AS preschool, preschool.id as preschool_form_id FROM students LEFT JOIN ( SELECT RANK() OVER (PARTITION BY student_id ORDER BY super_rank DESC, whencreated DESC) AS rank, id, super_rank, response_type, student_id, approval, whencreated FROM ( SELECT CASE WHEN u_fb_form_response.response_type = 'G' AND u_fb_form_response.approval = 1 THEN 1 ELSE 0 END AS super_rank, u_fb_form_response.id, u_fb_form_response.response_type, u_fb_form_response.student_id, u_fb_form_response.approval, u_fb_form_response.whencreated FROM u_fb_form_response WHERE u_fb_form_response.u_fb_form_id = 184243 ) ) preschool ON students.id = preschool.student_id AND preschool.rank = 1 WHERE students.enroll_status in (-1,0) AND students.Sched_NextYearGrade IN (-2,-1) ~[if.is.a.school] AND students.next_school=~(curschoolid) [/if] ) , max_submission AS ( SELECT u_fb_form_response.student_id, TO_CHAR(MAX(u_fb_form_response.whencreated), 'MM/DD/YYYY HH' || CHR(58) || 'MI am') AS lastest_submission FROM u_fb_form_response WHERE u_fb_form_response.response_type in ('G','S') GROUP BY u_fb_form_response.student_id ) SELECT student_form.dcid, student_form.student_number, student_form.lastfirst, student_form.current_grade, student_form.next_grade, REGEXP_SUBSTR(u_fb_form_element_choice.label, '[^|]+') as pk_program, CASE WHEN student_form.preschool = 'Approved' THEN '#198754' WHEN student_form.preschool = 'Pending' THEN '#ffc107' WHEN student_form.preschool = 'Rejected' THEN '#dc3545' WHEN student_form.preschool = 'Not Submitted' THEN '#6f42c1' WHEN student_form.preschool = 'Admin Submitted' THEN '#0d6efd' ELSE '' END as preschool_color, student_form.preschool, max_submission.lastest_submission, current_school.name AS current_school, next_school.name AS next_school FROM student_form LEFT JOIN u_student_additional_info ON student_form.dcid = u_student_additional_info.studentsdcid LEFT JOIN schools current_school ON student_form.current_school = current_school.school_number LEFT JOIN schools next_school ON student_form.next_school = next_school.school_number LEFT JOIN max_submission ON student_form.studentid = max_submission.student_id LEFT JOIN u_fb_form_response ON student_form.studentid = u_fb_form_response.student_id LEFT JOIN u_fb_form_response_detail ON u_fb_form_response.id = u_fb_form_response_detail.u_fb_form_response_id LEFT JOIN u_fb_form_response_element ON u_fb_form_response_detail.u_fb_form_response_element_id = u_fb_form_response_element.id LEFT JOIN u_fb_form_element_choice ON u_fb_form_response_element.U_FB_FORM_ELEMENT_ID = u_fb_form_element_choice.u_fb_form_element_id AND u_fb_form_response_detail.response_value = u_fb_form_element_choice.value WHERE u_fb_form_response_detail.u_fb_form_response_id = student_form.preschool_form_id AND (u_fb_form_response_detail.response_value = u_student_additional_info.pk_program) OR u_student_additional_info.pk_program IS NULL ORDER BY student_form.lastfirst; ;] [/tlist_sql]
