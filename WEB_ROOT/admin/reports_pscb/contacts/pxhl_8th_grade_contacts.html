AS CONTACT_CITY, CONTACT_DETAILS.CODE AS CONTACT_STATE, CONTACT_DETAILS.POSTALCODE AS CONTACT_ZIP FROM STUDENTS INNER JOIN STUDENTCONTACTASSOC ON STUDENTS.DCID = STUDENTCONTACTASSOC.STUDENTDCID INNER JOIN STUDENTCONTACTDETAIL ON STUDENTCONTACTDETAIL.STUDENTCONTACTASSOCID = STUDENTCONTACTASSOC.STUDENTCONTACTASSOCID LEFT JOIN OriginalContactMap ON STUDENTCONTACTASSOC.STUDENTCONTACTASSOCID = OriginalContactMap.STUDENTCONTACTASSOCID INNER JOIN CODESET ON STUDENTCONTACTDETAIL.RELATIONSHIPTYPECODESETID= CODESET.CODESETID INNER JOIN PERSON ON STUDENTCONTACTASSOC.PERSONID = PERSON.ID INNER JOIN SCHOOLS ON STUDENTS.SCHOOLID = SCHOOLS.SCHOOL_NUMBER INNER JOIN (SELECT PERSON.DCID, PERSON.ID, HOME_PHONE.PhoneNumber as HOME_PHONE, MOBILE_PHONE.PhoneNumber as MOBILE_PHONE, WORK_PHONE.PhoneNumber as WORK_PHONE, WORK_PHONE.PHONENUMBEREXT as WORK_PHONE_EXT, SINGLE_EMAIL.EMAILADDRESS, SINGLE_MAILING_ADDRESS.STREET, SINGLE_MAILING_ADDRESS.CITY, SINGLE_MAILING_ADDRESS.CODE, SINGLE_MAILING_ADDRESS.POSTALCODE FROM PERSON LEFT JOIN ( SELECT PERSON.DCID, PERSON.ID, PhoneNumber.PhoneNumber FROM PERSON INNER JOIN PersonPhoneNumberAssoc ON PERSON.ID = PersonPhoneNumberAssoc.PersonID INNER JOIN PhoneNumber ON PersonPhoneNumberAssoc.PhoneNumberID = PhoneNumber.PhoneNumberID INNER JOIN ( SELECT PERSON.DCID, PERSONID, min(PersonPhoneNumberAssoc.PhoneNumberPriorityOrder) minPhoneNumberPriorityOrder FROM PERSON INNER JOIN PersonPhoneNumberAssoc ON PERSON.ID = PersonPhoneNumberAssoc.PersonID INNER JOIN PhoneNumber ON PersonPhoneNumberAssoc.PhoneNumberID = PhoneNumber.PhoneNumberID WHERE PersonPhoneNumberAssoc.PhoneTypeCodeSetID = 13 GROUP BY PERSON.DCID, PERSONID ) SINGLE_MOBILE_NUMBER ON PERSON.DCID = SINGLE_MOBILE_NUMBER.DCID AND SINGLE_MOBILE_NUMBER.minPhoneNumberPriorityOrder = PersonPhoneNumberAssoc.PhoneNumberPriorityOrder AND PersonPhoneNumberAssoc.PhoneTypeCodeSetID = 13 ) MOBILE_PHONE ON PERSON.DCID = MOBILE_PHONE.DCID LEFT JOIN ( SELECT PERSON.DCID, PERSON.ID, PhoneNumber.PhoneNumber FROM PERSON INNER JOIN PersonPhoneNumberAssoc ON PERSON.ID = PersonPhoneNumberAssoc.PersonID INNER JOIN PhoneNumber ON PersonPhoneNumberAssoc.PhoneNumberID = PhoneNumber.PhoneNumberID INNER JOIN ( SELECT PERSON.DCID, PERSONID, min(PersonPhoneNumberAssoc.PhoneNumberPriorityOrder) minPhoneNumberPriorityOrder FROM PERSON INNER JOIN PersonPhoneNumberAssoc ON PERSON.ID = PersonPhoneNumberAssoc.PersonID INNER JOIN PhoneNumber ON PersonPhoneNumberAssoc.PhoneNumberID = PhoneNumber.PhoneNumberID WHERE PersonPhoneNumberAssoc.PhoneTypeCodeSetID = 12 GROUP BY PERSON.DCID, PERSONID ) SINGLE_HOME_NUMBER ON PERSON.DCID = SINGLE_HOME_NUMBER.DCID AND SINGLE_HOME_NUMBER.minPhoneNumberPriorityOrder = PersonPhoneNumberAssoc.PhoneNumberPriorityOrder AND PersonPhoneNumberAssoc.PhoneTypeCodeSetID = 12 ) HOME_PHONE ON PERSON.DCID = HOME_PHONE.DCID LEFT JOIN ( SELECT PERSON.DCID, PERSON.ID, PhoneNumber.PhoneNumber, PhoneNumber.PHONENUMBEREXT FROM PERSON INNER JOIN PersonPhoneNumberAssoc ON PERSON.ID = PersonPhoneNumberAssoc.PersonID INNER JOIN PhoneNumber ON PersonPhoneNumberAssoc.PhoneNumberID = PhoneNumber.PhoneNumberID INNER JOIN ( SELECT PERSON.DCID, PERSONID, min(PersonPhoneNumberAssoc.PhoneNumberPriorityOrder) minPhoneNumberPriorityOrder FROM PERSON INNER JOIN PersonPhoneNumberAssoc ON PERSON.ID = PersonPhoneNumberAssoc.PersonID INNER JOIN PhoneNumber ON PersonPhoneNumberAssoc.PhoneNumberID = PhoneNumber.PhoneNumberID WHERE PersonPhoneNumberAssoc.PhoneTypeCodeSetID = 15 GROUP BY PERSON.DCID, PERSONID ) SINGLE_WORK_NUMBER ON PERSON.DCID = SINGLE_WORK_NUMBER.DCID AND SINGLE_WORK_NUMBER.minPhoneNumberPriorityOrder = PersonPhoneNumberAssoc.PhoneNumberPriorityOrder AND PersonPhoneNumberAssoc.PhoneTypeCodeSetID = 15 ) WORK_PHONE ON PERSON.DCID = WORK_PHONE.DCID LEFT JOIN ( SELECT PERSON.DCID, PERSONID, LISTAGG(EMAILADDRESS.EMAILADDRESS,Chr(59)) WITHIN GROUP (ORDER BY EMAILADDRESS.EMAILADDRESS ASC) EMAILADDRESS FROM PERSON LEFT JOIN PERSONEMAILADDRESSASSOC ON PERSON.ID = PERSONEMAILADDRESSASSOC.PERSONID LEFT JOIN EMAILADDRESS ON PERSONEMAILADDRESSASSOC.EMAILADDRESSID = EMAILADDRESS.EMAILADDRESSID GROUP BY PERSON.DCID, PERSONID, PERSON.FIRSTNAME, PERSON.LASTNAME) SINGLE_EMAIL ON PERSON.DCID = SINGLE_EMAIL.DCID LEFT JOIN ( SELECT PERSON.DCID, PERSON.ID, PERSONADDRESS.STREET, PERSONADDRESS.CITY, STATECODE.CODE ,PERSONADDRESS.POSTALCODE FROM PERSON INNER JOIN PERSONADDRESSASSOC ON PERSON.ID = PERSONADDRESSASSOC.PERSONID INNER JOIN PERSONADDRESS ON PERSONADDRESSASSOC.PERSONADDRESSID = PERSONADDRESS.PERSONADDRESSID INNER JOIN CODESET STATECODE ON PERSONADDRESS.STATESCODESETID = STATECODE.CODESETID INNER JOIN (SELECT PERSON.DCID, PERSONID, min(PERSONADDRESSASSOC.ADDRESSPRIORITYORDER) ADDRESSPRIORITYORDER FROM PERSON INNER JOIN
